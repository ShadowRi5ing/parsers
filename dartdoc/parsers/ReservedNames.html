ReservedNames class 

class ReservedNames {
 Map<String, Parser<String>> _map;
 ReservedNames._(this._map);
 Parser<String> operator[](String key) {
   final res = _map[key];
   if (res == null) throw "$key is not a reserved name";
   else return res;
 }
}
